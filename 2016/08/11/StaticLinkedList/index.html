摘要：
　　静态链表是指用数组描述的链表，也称为游标实现法，下面给出静态链表的java语言实现。

}

定义一个静态链表对象及基本方法

package datastructure.linkedlist;

public class StaticLinkedList {

    //定义链表最大长度为1000
    public static  int MAXSIZE = 1000;

    //存放结点的数组
    NodeforStaticLinkedList [] nodelist = new NodeforStaticLinkedList [MAXSIZE];
    //链表的长度，即数据元素的长度
    int length;

    public NodeforStaticLinkedList[] getNodelist() {
        return this.nodelist;
    }

    public void setNodelist(NodeforStaticLinkedList[] nodelist) {
        this.nodelist = nodelist;
    }

    public int getLength() {
        return length;
    }

    public void setLength(int length) {
        this.length = length;
    }

    public StaticLinkedList() {
        //nodelist[0].cur = 0;
        //nodelist[MAXSIZE-1].cur = 0;
        initList();

    }


    public boolean initList() {

        for(int i = 0; i < MAXSIZE-1; i++) {
            nodelist[i] = new NodeforStaticLinkedList();
            nodelist[i].cur = i+1;
        }
        nodelist[MAXSIZE-1] = new NodeforStaticLinkedList();
        nodelist[MAXSIZE-1].cur = 0;
        length = 0;
        return true;
    }

    //分配结点
    public int Malloc_SSL() {
        int i = nodelist[0].cur;
        if(!(nodelist[0].cur == 0)) {
            nodelist[0].cur = nodelist[i].cur;
        }        
        return i;
    }

    //在静态链表中的第i个元素之前插入新元素
    public boolean ListInsert(StaticLinkedList L, int i, int e) {

        if(i < 1 || i > L.getLength() + 1) {
            System.out.println("ILLEGAL INPUT");
            return false;
        }

        //获取待分配结点的游标     
        int j = L.Malloc_SSL();;
        int k = MAXSIZE-1;
        if(!(j == 0)) {

            L.nodelist[j].data = e;

            for(int l = 1; l <= i - 1; l++) {
                k = L.nodelist[k].cur;
            }

        L.nodelist[j].cur = L.nodelist[k].cur ;
        L.nodelist[k].cur = j;
        L.setLength(L.getLength() + 1);

        return true;

        }

        return false;

    }

}

测试代码

package datastructure.linkedlist;

public class test {
    //定义链表最大长度为1000
    public static  int MAXSIZE = 1000;

    public static void main(String[] args) {
         StaticLinkedList testList = new StaticLinkedList();

         testList.initList();
         System.out.println("初始化后头结点的游标指向的备用结点在：" + testList.nodelist[0].cur);
         System.out.println("初始化后尾结点的游标指向的第一个元素在：" + testList.nodelist[MAXSIZE-1].cur);
         testList.ListInsert(testList, 1, 10);
         testList.ListInsert(testList, 2, 20);
         testList.ListInsert(testList, 3, 40);
         testList.ListInsert(testList, 4, 50);
         testList.ListInsert(testList, 5, 60);
         System.out.println("连续插入元素后头结点的游标指向的备用结点在：" + testList.nodelist[0].cur);
         System.out.println("连续插入元素后尾结点的游标指向的第一个元素在：" + testList.nodelist[MAXSIZE-1].cur);
         int l = testList.length;
         System.out.println("插入位置3的元素前表长：" + l);
         System.out.println("插入前头结点的游标指向的备用结点在：" + testList.nodelist[0].cur);
         for(int i = 1; i <= l; i++) {
             System.out.print(i + " "  +  testList.nodelist[i].data + " "+  testList.nodelist[i].cur  + " ");
         }

         testList.ListInsert(testList, 3, 30);
         System.out.println("\n" + "插入位置3的元素后头结点的游标指向的备用结点在：" + testList.nodelist[0].cur);
         System.out.println("插入位置3的元素后尾结点的游标指向的第一个元素在：" + testList.nodelist[MAXSIZE-1].cur);
         int l1 = testList.length;
         System.out.println("插入位置3的元素后表长：" + l1);
         System.out.println("插入位置3的元素后头结点的游标指向的备用结点在：" + testList.nodelist[0].cur);
         for(int i = 1; i <= l1; i++) {
             System.out.print(i + " "  +  testList.nodelist[i].data + " "+  testList.nodelist[i].cur + " " );
         }    

         testList.ListInsert(testList, 1, 100);
         System.out.println("\n" + "插入位置3的元素后尾结点的游标指向的第一个元素在：" + testList.nodelist[MAXSIZE-1].cur);

    }

}
